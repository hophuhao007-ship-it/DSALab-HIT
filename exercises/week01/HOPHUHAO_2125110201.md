#include <iostream>

struct Date {
    int d, m, y;
};

// ================= LOAD LOGIN =================
int loadLogin() {

    ifstream file("login.txt");

    int x = 0;

    if (file.is_open()) {

        file >> x;

        file.close();
    }

    return x;
}

// ================= SAVE LOGIN =================
void saveLogin(int x) {

    ofstream file("login.txt");

    file << x;

    file.close();
}

// ================= INPUT DATE =================
Date inputDate(string msg) {

    Date a;
    char x;

    cout << msg;

    cin >> a.d >> x >> a.m >> x >> a.y;

    return a;
}

// ================= VALID =================
bool valid(Date a) {

    if (a.y < 1)
        return false;

    if (a.m < 1 || a.m > 12)
        return false;

    if (a.d < 1 || a.d > 31)
        return false;

    return true;
}

// ================= CON GIAP =================
string zodiacAnimal(int y) {

    string z[12] = {
        "Ty", "Suu", "Dan", "Mao",
        "Thin", "Ti", "Ngo", "Mui",
        "Than", "Dau", "Tuat", "Hoi"
    };

    int idx = (y - 4) % 12;

    if (idx < 0)
        idx += 12;

    return z[idx];
}

// ================= CUNG =================
string zodiacSign(int d, int m) {

    if ((m == 3 && d >= 21) || (m == 4 && d <= 19))
        return "Bach Duong";

    if ((m == 4 && d >= 20) || (m == 5 && d <= 20))
        return "Kim Nguu";

    if ((m == 5 && d >= 21) || (m == 6 && d <= 20))
        return "Song Tu";

    if ((m == 6 && d >= 21) || (m == 7 && d <= 22))
        return "Cu Giai";

    if ((m == 7 && d >= 23) || (m == 8 && d <= 22))
        return "Su Tu";

    if ((m == 8 && d >= 23) || (m == 9 && d <= 22))
        return "Xu Nu";

    if ((m == 9 && d >= 23) || (m == 10 && d <= 22))
        return "Thien Binh";

    if ((m == 10 && d >= 23) || (m == 11 && d <= 21))
        return "Bo Cap";

    if ((m == 11 && d >= 22) || (m == 12 && d <= 21))
        return "Nhan Ma";

    if ((m == 12 && d >= 22) || (m == 1 && d <= 19))
        return "Ma Ket";

    if ((m == 1 && d >= 20) || (m == 2 && d <= 18))
        return "Bao Binh";

    return "Song Ngu";
}

// ================= MAY MAN =================
int calcLuck(Date d) {

    return (d.d * 3 + d.m * 7 + d.y) % 101;
}

// ================= SO MAY MAN =================
int luckyNumber(Date d) {

    return (d.d + d.m + d.y) % 9 + 1;
}

// ================= SO THICH =================
string hobby(string sign, string gender) {

    if (gender == "nam") {

        if (sign == "Su Tu")
            return "The thao, lanh dao";

        if (sign == "Bo Cap")
            return "Cong nghe, kham pha";

        return "Game, du lich";
    }

    else {

        if (sign == "Kim Nguu")
            return "Am nhac, mua sam";

        if (sign == "Song Ngu")
            return "Nghe thuat, chup anh";

        return "Doc sach, cafe";
    }
}

// ================= SONG BAO NHIEU NGAY =================
long long daysAlive(Date a, Date now) {

    long long years = now.y - a.y;
    long long months = now.m - a.m;
    long long days = now.d - a.d;

    return years * 365 + months * 30 + days;
}

// ================= LOVE SCORE =================
int loveScore(Date a, Date b) {

    int x = (a.d + a.m + a.y) % 100;
    int y = (b.d + b.m + b.y) % 100;

    return 100 - abs(x - y);
}

// ================= PROGRESS BAR =================
void progressBar(int exp) {

    int bars = exp / 10;

    cout << "EXP BAR        : [";

    for (int i = 0; i < bars; i++)
        cout << "#";

    for (int i = bars; i < 10; i++)
        cout << "-";

    cout << "]\n";
}
