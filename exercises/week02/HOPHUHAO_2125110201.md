#include <iostream>
using namespace std;

int main() {

    // ==============================
    // KHAI BAO SO PHAN TU
    // ==============================
    int n;

    cout << "Nhap so phan tu: ";
    cin >> n;

    // ==============================
    // KHAI BAO MANG
    // ==============================
    int a[100];

    // ==============================
    // NHAP DU LIEU CHO MANG
    // ==============================
    for (int i = 0; i < n; i++) {

        cout << "Nhap a[" << i << "] = ";
        cin >> a[i];
    }

    // ==============================
    // KHOI TAO BIEN
    // ==============================
    int sum = 0;

    // Gan min va max bang phan tu dau tien
    int minVal = a[0];
    int maxVal = a[0];

    // ==============================
    // XU LY DU LIEU
    // ==============================
    for (int i = 0; i < n; i++) {

        // Tinh tong
        sum += a[i];

        // Tim min
        if (a[i] < minVal) {
            minVal = a[i];
        }

        // Tim max
        if (a[i] > maxVal) {
            maxVal = a[i];
        }
    }

    // ==============================
    // TINH TRUNG BINH CONG
    // ==============================
    float avg = (float)sum / n;

    // ==============================
    // XUAT KET QUA
    // ==============================
    cout << "\n===== KET QUA =====\n";

    cout << "Tong = " << sum << endl;
    cout << "Gia tri nho nhat = " << minVal << endl;
    cout << "Gia tri lon nhat = " << maxVal << endl;
    cout << "Trung binh cong = " << avg << endl;

    return 0;
}
